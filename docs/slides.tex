\begin{frame}{Problem: Attention Computation}
    \begin{itemize}
        \item Attention is the core of Transformer models: $O = \text{softmax}\left(\frac{QK^T}{\sqrt{d}}\right)V$.
        \item It is memory bandwidth bound, not compute-bound.
        \item The attention matrix $S = QK^T$ has a size of $O(N^2)$, where $N$ is the sequence length.
        \item Moving $O(N^2)$ intermediate data back and forth from \textbf{Global Memory (HBM)} causes a severe performance bottleneck.
    \end{itemize}
\end{frame}

    \vspace{0.1cm}
    \footnotesize
    \textbf{Kernel Fusion \& Tiling:} Matrices load in blocks (tiles) into fast \textbf{SRAM}. Attention and \textit{online} softmax happen on-chip. The $O(N^2)$ matrix $S$ is never materialized to HBM, resulting in $O(N)$ memory bounds.
\end{frame}

        \node[font=\footnotesize\bfseries] at (0, 0.7) {Paged Attention KV Cache};
        
        % Logical Page Table
        \node[draw, rectangle, minimum width=2.5cm, minimum height=1.5cm, align=left, font=\scriptsize, fill=yellow!10] (pt) at (-1.5, -0.6) {\textbf{Page Table}\\L-pg 0 $\rightarrow$ P-pg 2\\L-pg 1 $\rightarrow$ P-pg 0};

        \node[cell, used] (p0) at (2.5, 0) {P-pg 0};
        \node[cell, unused] (p1) at (2.5, -0.7) {P-pg 1};
        \node[cell, used] (p2) at (2.5, -1.4) {P-pg 2};
        \node[cell, unused] (p3) at (2.5, -2.1) {P-pg 3};
        
        \draw[arrow] (-0.2, -0.4) -- (p2.west);
        \draw[arrow] (-0.2, -0.8) -- (p0.west);
        
    \end{tikzpicture}
    \end{center}
    
    \vspace{0.1cm}
    \footnotesize
    \textbf{Solution:} OS paging concepts for the KV Cache. Memory is dynamically allocated in non-contiguous \textbf{blocks} instead of continuous chunks. Thread blocks resolve logical KV addresses seamlessly via the \textbf{Page Table}.
\end{frame}

\begin{frame}{End-to-End Integration in GPT-2}
    \begin{itemize}
        \item We integrated our custom \textbf{Flash Attention} and \textbf{Paged Attention} CUDA kernels directly into the HuggingFace GPT-2 model.
        \item \textbf{Prefill Phase:} Uses the Flash Attention kernel to process the initial prompt.
        \item \textbf{Decode Phase:} Uses the Paged Attention kernel to process generated tokens sequentially.
    \end{itemize}
